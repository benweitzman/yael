-- | Lifted version of 'STM.atomically'
--
-- @since 0.2.1.0

-- | Renamed 'STM.retry' for unqualified export

-- @since 0.2.1.0

retrySTM = STM.retry

-- | Renamed 'STM.check' for unqualified export

-- @since 0.2.1.0
